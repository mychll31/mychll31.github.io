setup git repository

  1. sudo apt-get install git-core
  2. ssh-keygen -t rsa
  3. cat ~/.ssh/id_rsa.pub | ssh @ "mkdir -p ~/.ssh && cat >> ~/.ssh/"
  4. mkdir -p /home/swapnil/project-1.git //create folder
  5. cd to [no.4]
  6. git init --bare path/to/project-1.git 
Clone: 

git clone ssh://@/path/to/project-1.git
